CREATE TABLE UserProfiles (
	ID INT NOT NULL PRIMARY KEY IDENTITY(1,1),
	UserID NVARCHAR(450) NOT NULL FOREIGN KEY REFERENCES AspNetUsers(Id),

	Email NVARCHAR (256),
	HomeState NVARCHAR (50),
